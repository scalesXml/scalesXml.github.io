220    * the element, when the resulting list contains a Done
221    * then the list of Dones is returned.
222    *
223    * One can use tuples or pattern matching to discern which
224    * of the original lists iteratees have matched.
225    *
226    * Due to the return type of this iteratee all items
227    * in the list must return the same type and must return
228    * both an A and an IterV with the same interface to continue
229    * with the next and maintain state.
230    *
231    *
232    * In the case of EOF, an empty list is returned
233    */
234   def onDone[E,A](originalList : List[ResumableIter[E,A]]) : ResumableIterList[E,A] = {
235
236     def step(l : List[ResumableIter[E,A]])(s: Input[E]): ResumableIterList[E,A] =
237       s(el = e => {
238   var res : List[A] = Nil
239   var newl : List[ ResumableIter[E,A] ] = Nil
240
241   @inline def add( k : (scalaz.Input[E]) => scalaz.IterV[E,(A, scalaz.IterV[E, _])]) {
242     val d = k(El(e))
243     newl = d :: newl
244     d match {
245       case Done(x, _) => res = x._1 :: res
246       case _ => ()
247     }
248   }
249
250   val i = l.iterator
251   while(i.hasNext) {
252     i.next match {
253     // safety first
254     case iter@Done(e1, _) =>
255       // are we EOF, in which case remove
256       if (isEOF(iter)) Nil
257       else {
258         if (isEmpty(iter))
259     // feed back the continuation
260     // this is where we hope that the users don't
261     // break on purpose :-()
262     e1._2 match {
263       case i @ Cont(k) => add(k.asInstanceOf[(scalaz.Input[E]) => scalaz.IterV[E,(A, scalaz.IterV[E, _])]])
264       case _ => error("Continuation can only be a Cont")
265     }
266         else error("Can only handle EOF or Empty for Done")
267     
268       }
269     case iter@Cont(k) => add(k)
270     }
271   }
272         
273   if (res.isEmpty)
274     Cont(step(newl))
275   else
276     Done((res, Cont(step(newl))), IterV.Empty[E])
277   },
278         empty = Cont(step(l)),
279         eof = Done((Nil, Cont(step(l))), EOF[E]))
280     Cont(step(originalList))
281   }
