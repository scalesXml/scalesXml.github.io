3 /**
4 * Very simple logger, matches scala more.  Each function returns true if the message would be logged.
5 *
6 * Each function takes an optional second parameter of a throwable.
7 *
8 * Implementations need only implement the log function.
9 */
10 trait Logger {
11
12   def trace(msg: => String, throwable: => Throwable): Boolean = log(Trace, msg, Some(throwable))
13   def debug(msg: => String, throwable: => Throwable): Boolean = log(Debug, msg, Some(throwable))
14   def info(msg: => String, throwable: => Throwable): Boolean = log(Info, msg, Some(throwable))
15   def warn(msg: => String, throwable: => Throwable): Boolean = log(Warn, msg, Some(throwable))
16   def error(msg: => String, throwable: => Throwable): Boolean = log(Error, msg, Some(throwable))
17
18   def trace(msg: => String): Boolean = log(Trace, msg, None)
19   def debug(msg: => String): Boolean = log(Debug, msg, None)
20   def info(msg: => String): Boolean = log(Info, msg, None)
21   def warn(msg: => String): Boolean = log(Warn, msg, None)
22   def error(msg: => String): Boolean = log(Error, msg, None)
23
24   /**
25    * Special one, logs against some specific logging level.  Wouldn't recommend using it,
26    * just use a special named logger instead.
27    */
28   def log(level: LogLevel, msg: => String, throwable: => Option[Throwable] = None): Boolean
29 }

48 /**
49 * Default implementation of a logger, for some reaason if its not
50 * excluded (via private) or we change the type to no longer inherit
51 * from logger we get a java.lang.UnsupportedOperationException: Position.Point
52 * exception
53 */
54 private[utils] class Slf4jLogger(logger : org.slf4j.Logger) extends Logger {
55   def log(level: LogLevel, msg: => String, throwable: => Option[Throwable]): Boolean = {
56       level match {
57         case Trace => if (logger.isTraceEnabled()) {if (throwable.isDefined) logger.trace(msg, throwable.get) else logger.trace(msg); true} else false
58         case Debug => if (logger.isDebugEnabled()) {if (throwable.isDefined) logger.debug(msg, throwable.get) else logger.debug(msg); true} else false
59         case Info => if (logger.isInfoEnabled()) {if (throwable.isDefined) logger.info(msg, throwable.get) else logger.info(msg); true} else false
60         case Warn => if (logger.isWarnEnabled()) {if (throwable.isDefined) logger.warn(msg, throwable.get) else logger.warn(msg); true} else false
61         case Error => if (logger.isErrorEnabled()) {if (throwable.isDefined) logger.error(msg, throwable.get) else logger.error(msg); true} else false
62       }
63     }
64 }
